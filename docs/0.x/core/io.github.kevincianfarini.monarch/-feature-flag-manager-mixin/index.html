interface FeatureFlagManagerMixin

A supplement to MixinFeatureFlagManager that allows extension via FeatureFlagManagerMixin.currentValueOfOrNull. Implementations of this interface can opt to handle 1 or more FeatureFlag types. For example, if we want to add a mixin that supports JSON, we could do so with the following code.

class JsonFeatureFlagMixin : FeatureFlagMixin {

  override suspend fun <T : FeatureFlagOption> currentValueForOrNull(
    flag: FeatureFlag<T>,
    store: FeatureFlagDataStore,
  ): T? = when (flag) {
    is SomeJsonFeatureFlag<T> -> decodeJson(store.getString(flag.key))
    else -> null
  }

  abstract suspend fun <T : FeatureFlagOption> decodeJson(flag: FeatureFlag<T>): T
}
